## Overview

In this demo, you'll learn how to:

- Create a compressed token mint
- Mint compressed tokens to an account
- Transfer compressed tokens between accounts
- Query compressed token balances

## Creating a Compressed Token Mint

First, let's create a new compressed token mint:

```typescript
import { createMint } from "@lightprotocol/zk-compression-sdk";

async function createCompressedMint() {
  const mint = await createMint({
    connection,
    payer,
    mintAuthority: payer.publicKey,
    decimals: 9, // Standard token decimals
  });

  console.log("‚úÖ Created compressed mint:", mint.toBase58());
  return mint;
}
```

## Minting Compressed Tokens

Now let's mint some compressed tokens to your account:

```typescript
import { mintTo } from "@lightprotocol/zk-compression-sdk";

async function mintCompressedTokens(mint: PublicKey, amount: number) {
  const signature = await mintTo({
    connection,
    payer,
    mint,
    destination: payer.publicKey,
    authority: payer,
    amount: amount * 10 ** 9, // Convert to lamports
  });

  console.log("‚úÖ Minted tokens:", signature);
  return signature;
}
```

## Transferring Compressed Tokens

Transfer compressed tokens to another account:

```typescript
import { transfer } from "@lightprotocol/zk-compression-sdk";
import { Keypair } from "@solana/web3.js";

async function transferCompressedTokens(mint: PublicKey, amount: number) {
  // Create a recipient account
  const recipient = Keypair.generate();

  const signature = await transfer({
    connection,
    payer,
    mint,
    from: payer.publicKey,
    to: recipient.publicKey,
    authority: payer,
    amount: amount * 10 ** 9,
  });

  console.log("‚úÖ Transferred tokens:", signature);
  console.log("üìß Recipient:", recipient.publicKey.toBase58());

  return { signature, recipient };
}
```

## Querying Balances

Check compressed token balances:

```typescript
import { getCompressedTokenBalance } from "@lightprotocol/zk-compression-sdk";

async function checkBalance(mint: PublicKey, owner: PublicKey) {
  const balance = await getCompressedTokenBalance({
    connection,
    mint,
    owner,
  });

  console.log(`üí∞ Balance: ${balance / 10 ** 9} tokens`);
  return balance;
}
```

## Complete Example

Here's a complete example that puts it all together:

```typescript
import {
  createMint,
  mintTo,
  transfer,
  getCompressedTokenBalance,
} from "@lightprotocol/zk-compression-sdk";
import { Connection, Keypair } from "@solana/web3.js";

async function compressedTokenDemo() {
  try {
    // 1. Create a compressed mint
    console.log("üè≠ Creating compressed mint...");
    const mint = await createMint({
      connection,
      payer,
      mintAuthority: payer.publicKey,
      decimals: 9,
    });

    // 2. Mint tokens to your account
    console.log("ü™ô Minting 1000 tokens...");
    await mintTo({
      connection,
      payer,
      mint,
      destination: payer.publicKey,
      authority: payer,
      amount: 1000 * 10 ** 9,
    });

    // 3. Check your balance
    const yourBalance = await getCompressedTokenBalance({
      connection,
      mint,
      owner: payer.publicKey,
    });
    console.log(`üí∞ Your balance: ${yourBalance / 10 ** 9} tokens`);

    // 4. Create recipient and transfer tokens
    const recipient = Keypair.generate();
    console.log("üì§ Transferring 100 tokens...");

    await transfer({
      connection,
      payer,
      mint,
      from: payer.publicKey,
      to: recipient.publicKey,
      authority: payer,
      amount: 100 * 10 ** 9,
    });

    // 5. Check balances after transfer
    const yourNewBalance = await getCompressedTokenBalance({
      connection,
      mint,
      owner: payer.publicKey,
    });

    const recipientBalance = await getCompressedTokenBalance({
      connection,
      mint,
      owner: recipient.publicKey,
    });

    console.log(`üí∞ Your new balance: ${yourNewBalance / 10 ** 9} tokens`);
    console.log(`üí∞ Recipient balance: ${recipientBalance / 10 ** 9} tokens`);

    console.log("üéâ Demo completed successfully!");
  } catch (error) {
    console.error("‚ùå Demo failed:", error);
  }
}

// Run the demo
compressedTokenDemo();
```

- **Familiar APIs**: Same developer experience as regular Solana tokens
- **Full Functionality**: All standard token operations work seamlessly
- **Scalability**: Can handle millions of token holders efficiently
